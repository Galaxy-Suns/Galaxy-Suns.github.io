'''函数参数定义_个数可变的位置形参 个数可变的关键字形参'''

# 个数可变的位置形参

def fun(*args):  # 函数定义时的个数可变的位置参数，在形参前加 * ，使形参为一个时实参可以为多个或一个，实参以元组传到形参中
    print(args)
    print(args[0])

fun(10)
fun(10,30)
fun(20,405,50)

# 个数可变的关键字形参

def fun1(**args):  # 函数定义时的个数可变的关键字参数，在形参前加 ** ，使形参为一个时实参可以用关键字赋值多个或一个，实参以字典传到形参中
    print(args)

fun1(a=10)
fun1(a=10,b=30,c=40) # {'a': 10, 'b': 30, 'c': 40}

print('hello','world','java')

'''
由于函数 print() 的个数可变的位置形参 *args 
所以可以输出多个值
'''
